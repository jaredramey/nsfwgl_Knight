#version 410

layout(location=0) in vec4 Position;
layout(location=1) in vec4 Normal;
layout(location=2) in vec4 Tangent;
layout(location=3) in vec2 TexCoord;

out vec4 vPosition;
out vec4 vNormal;
out vec2 vTexCoord;

uniform mat4 View;
uniform mat4 Projection;
uniform mat4 ProjectionView;
uniform mat4 Model;

uniform sampler2D NormalTex;

void main()
{
	vPosition =  View * Model * Position;	
	vTexCoord = TexCoord;
	gl_Position =  Projection * View * Model * Position;
	
	vec4 NormalSample = vec4( normalize( texture(NormalTex, TexCoord).xyz * 2 - 1), 0);

	
	vec4 dBiTangent = vec4( normalize(cross(Normal.xyz, Tangent.xyz) ), 0);
	
	mat4 TBN = Model * mat4(Tangent, dBiTangent, Normal, vec4(0, 0, 0, 1));

	vNormal = View * TBN * NormalSample;
}
